BACK PROPAGATION

basicamente:

termo de erro: (y - y_hat) * derivada do valor que chega (sigmoid_prime do W + b do hidden anterior pra esse output layer, em que W seria a sigmoid de x * conexao)

erro anterior (hidden):
	termo de erro da frente * W da conexas da frente com layer atual (ponderada pelo W, se preferir) * derivada do valor que chega a esse layer...

e assim por diante

==========================================
SEMPRE:

	erro da frente * derivada do valor anterior ate o layer atual (que seria sigmoid_prime do x ao invés do sigmoid de x) * peso do layer atual para o proximo (weight)


	atualizacao peso durate o epoch:
		peso += learning rate * erro ate aquele layer * a (sendo a o valor da função de ativação do nó anterior... sem multiplicar pelo peso da aresta que havia antes)
	caso seja o primeiro layer dps do input, esse "a" será o valor da feature

	ao final do epoch, divida pelo numero de observações
==========================================

import numpy as np
from data_prep import features, targets, features_test, targets_test

np.random.seed(21)

def sigmoid(x):
    """
    Calculate sigmoid
    """
    return 1 / (1 + np.exp(-x))

def sigmoid_prime(x):
    """
    Calculate sigmoid derivative
    """
    return sigmoid(x) * (1 - sigmoid(x))

# Hyperparameters
n_hidden = 2  # number of hidden units
epochs = 100
learnrate = 0.05

n_records, n_features = features.shape
last_loss = None
# Initialize weights
weights_input_hidden = np.random.normal(scale=1 / n_features ** .5,
                                        size=(n_features, n_hidden))
weights_hidden_output = np.random.normal(scale=1 / n_features ** .5,
                                         size=n_hidden)

for e in range(epochs):
    del_w_input_hidden = np.zeros(weights_input_hidden.shape)
    del_w_hidden_output = np.zeros(weights_hidden_output.shape)
    for x, y in zip(features.values, targets):
        ## Forward pass ##
        # TODO: Calculate the output
        hidden_input = x.dot(weights_input_hidden)
        hidden_output = sigmoid(hidden_input)
        output = sigmoid(hidden_output.dot(weights_hidden_output))

        ## Backward pass ##
        # TODO: Calculate the network's prediction error
        error = y - output

        # TODO: Calculate error term for the output unit
        output_error_term = error * sigmoid_prime(hidden_output.dot(weights_hidden_output))
        
        ## propagate errors to hidden layer

        # TODO: Calculate the hidden layer's contribution to the error
        #hidden_error = sigmoid_prime(hidden_input)
        
        # TODO: Calculate the error term for the hidden layer
        hidden_error_term = output_error_term * sigmoid_prime(hidden_input) * weights_hidden_output
        
        # TODO: Update the change in weights
        del_w_hidden_output += output_error_term * hidden_output
        del_w_input_hidden += hidden_error_term * x[:, None]

    # TODO: Update weights  (don't forget to division by n_records or number of samples)
    weights_input_hidden += learnrate * del_w_input_hidden / n_records
    weights_hidden_output += learnrate * del_w_hidden_output / n_records

    # Printing out the mean square error on the training set
    if e % (epochs / 10) == 0:
        hidden_output = sigmoid(np.dot(x, weights_input_hidden))
        out = sigmoid(np.dot(hidden_output,
                             weights_hidden_output))
        loss = np.mean((out - targets) ** 2)

        if last_loss and last_loss < loss:
            print("Train loss: ", loss, "  WARNING - Loss Increasing")
        else:
            print("Train loss: ", loss)
        last_loss = loss

# Calculate accuracy on test data
hidden = sigmoid(np.dot(features_test, weights_input_hidden))
out = sigmoid(np.dot(hidden, weights_hidden_output))
predictions = out > 0.5
accuracy = np.mean(predictions == targets_test)
print("Prediction accuracy: {:.3f}".format(accuracy))
